// Function pointer example
typedef int (*operation)(int, int);

int add(int a, int b) { return a + b; }
int subtract(int a, int b) { return a - b; }

int calculate(int x, int y, operation op) {
return op(x, y);
}

printf("Result: %d\n", result);
return 0;
}

// Dynamic array with realloc
int *create_dynamic_array(size_t size) {
int *arr = (int*)malloc(size * sizeof(int));
if (!arr) {
fprintf(stderr, "Memory allocation failed\n");
return NULL;
}
return arr;
}

int *resize_array(int *arr, size_t new_size) {
int *new_arr = (int*)realloc(arr, new_size * sizeof(int));
if (!new_arr) {
free(arr);
return NULL;
}
return new_arr;
}

// Linked list node structure
typedef struct Node {
int data;
struct Node *next;
} Node;

// Insert at beginning
Node* insert_front(Node *head, int value) {
Node *new_node = (Node*)malloc(sizeof(Node));
new_node->data = value;
new_node->next = head;
return new_node;
}

// Free entire list
void free_list(Node *head) {
Node *temp;
while (head) {
temp = head;
head = head->next;
free(temp);
}
}

-- Subquery factoring (Common Table Expression)
WITH dept_totals AS (
SELECT department_id,
SUM(salary) as total_salary,
COUNT(*) as emp_count
FROM employees
GROUP BY department_id
)
SELECT d.department_name,
dt.total_salary,
dt.emp_count,
dt.total_salary / dt.emp_count as avg_salary
FROM departments d
JOIN dept_totals dt ON d.department_id = dt.department_id
WHERE dt.total_salary > 100000;

  • FIRST_VALUE() and LAST_VALUE()
  • NTILE() for data distribution
  • PARTITION BY clause
  • Window frame specifications
  • Moving averages and cumulative sums

-- Analytical functions example
SELECT
employee_id,
department_id,
salary,
ROW_NUMBER() OVER (PARTITION BY department_id ORDER BY salary DESC) as rank_in_dept,
LAG(salary) OVER (ORDER BY hire_date) as prev_emp_salary,
SUM(salary) OVER (PARTITION BY department_id) as dept_total,
AVG(salary) OVER (ORDER BY hire_date ROWS BETWEEN 2 PRECEDING AND CURRENT ROW) as moving_avg
FROM employees;

3. Hierarchical Queries

  • CONNECT BY clause

  • PRIOR operator

  • SYS_CONNECT_BY_PATH function
  • CONNECT_BY_ROOT operator

-- Hierarchical query for organizational structure
SELECT
LEVEL,
LPAD(' ', 2*(LEVEL-1)) || employee_name as org_chart,
manager_id,
SYS_CONNECT_BY_PATH(employee_name, '/') as path
FROM employees
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id
ORDER SIBLINGS BY employee_name;

Example Conversion:


-- Oracle CONNECT BY
SELECT LEVEL, employee_name
FROM employees
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id;

-- Converted to PostgreSQL Recursive CTE
WITH RECURSIVE emp_hierarchy AS (
SELECT employee_id, employee_name, manager_id, 1 as level
FROM employees
WHERE manager_id IS NULL
UNION ALL
SELECT e.employee_id, e.employee_name, e.manager_id, eh.level + 1
FROM employees e
JOIN emp_hierarchy eh ON e.manager_id = eh.employee_id
)
SELECT level, employee_name FROM emp_hierarchy;

#!/bin/bash
# Database Backup Automation Script

BACKUP_DIR="/opt/backups"
DB_LIST=("production_db" "staging_db" "analytics_db")
DATE=$(date +%Y%m%d_%H%M%S)
RETENTION_DAYS=7

for db in "${DB_LIST[@]}"; do
echo "Backing up $db..."
pg_dump $db | gzip > ${BACKUP_DIR}/${db}_${DATE}.sql.gz

if [ ${PIPESTATUS[0]} -eq 0 ]; then
echo "✓ $db backup successful"
else
echo "✗ $db backup failed" | mail -s "Backup Failed" admin@example.com
fi
done

# Clean old backups
find $BACKUP_DIR -name "*.sql.gz" -mtime +$RETENTION_DAYS -delete

#!/bin/bash
# Database Health Check Script

check_database_connections() {
local max_conn=$(psql -U postgres -t -c "SHOW max_connections;" | xargs)
local current=$(psql -U postgres -t -c "SELECT count(*) FROM pg_stat_activity;" | xargs)
local percent=$((current * 100 / max_conn))

echo "Connections: $current / $max_conn ($percent%)"

if [ $percent -gt 80 ]; then
echo "WARNING: High connection usage!" | mail -s "DB Alert" dba@example.com
fi
}

check_long_running_queries() {
psql -U postgres -c "
SELECT pid, now() - query_start as duration, query
FROM pg_stat_activity
WHERE state = 'active'
AND now() - query_start > interval '5 minutes';"
}

check_database_size() {
psql -U postgres -c "
SELECT datname, pg_size_pretty(pg_database_size(datname))
FROM pg_database ORDER BY pg_database_size(datname) DESC;"
}

// Task queue for database operations
Queue<String> taskQueue = new LinkedList<>();
taskQueue.offer("backup_database");
taskQueue.offer("optimize_tables");
taskQueue.offer("update_statistics");

// Process tasks
while (!taskQueue.isEmpty()) {
String task = taskQueue.poll();
System.out.println("Executing: " + task);
}

// PriorityQueue for priority-based processing
PriorityQueue<Integer> priorities = new PriorityQueue<>();
priorities.offer(3);
priorities.offer(1);
priorities.offer(2);
// Will process in order: 1, 2, 3

// Traditional iterator
Iterator<String> iterator = databases.iterator();
while (iterator.hasNext()) {
String db = iterator.next();
System.out.println(db);
}

// Enhanced for-loop
for (String db : databases) {
System.out.println(db);
}

// Stream API
databases.stream()
.filter(db -> db.startsWith("prod"))
.forEach(System.out::println);

// Database info class
class DatabaseInfo {
String name;
int size;

DatabaseInfo(String name, int size) {
this.name = name;
this.size = size;
}
}

List<DatabaseInfo> databases = new ArrayList<>();
databases.add(new DatabaseInfo("prod", 500));
databases.add(new DatabaseInfo("dev", 100));
databases.add(new DatabaseInfo("staging", 250));

// Sort by size
databases.sort(Comparator.comparingInt(db -> db.size));

// Sort by name
databases.sort(Comparator.comparing(db -> db.name));

// Reverse order
databases.sort(Comparator.comparingInt(db -> db.size).reversed());

#!/bin/bash

# If statement
if [ -f "/var/lib/postgresql/data/postgresql.conf" ]; then
echo "PostgreSQL config found"
else
echo "Config not found"
exit 1
fi

# If-elif-else
if [ $# -eq 0 ]; then
echo "No arguments provided"
elif [ $# -eq 1 ]; then
echo "One argument: $1"
else
echo "Multiple arguments: $@"
fi

# Case statement
case "$1" in
start)
echo "Starting database..."
;;
stop)
echo "Stopping database..."
;;
restart)
echo "Restarting database..."
;;
*)
echo "Usage: $0 {start|stop|restart}"
exit 1
;;
esac

# While loop - monitor process

while [ $counter -le 5 ]; do
echo "Check #$counter"
ps aux | grep postgres
sleep 5
((counter++))
done

# Read file line by line

#!/bin/bash

# Function definition
backup_database() {
local db_name=$1
local backup_dir=$2
local date=$(date +%Y%m%d_%H%M%S)

echo "Starting backup of $db_name..."
pg_dump $db_name > ${backup_dir}/${db_name}_${date}.sql

if [ $? -eq 0 ]; then
echo "Backup successful"
return 0
else
echo "Backup failed"
return 1
fi
}

# Function call
backup_database "production_db" "/opt/backups"

if [ $? -eq 0 ]; then
echo "All done!"
fi

8. Error Handling

  • Exit codes and status
  • Error checking
  • set -e for exit on error
  • set -u for undefined variables
  • Trap signals
  • Error logging
  • Graceful error handling

#!/bin/bash
set -e # Exit on error
set -u # Exit on undefined variable

# Error handling function
handle_error() {
echo "Error occurred in script at line $1"
echo "Cleaning up..."
# Cleanup code here
exit 1
}

# Trap errors
trap 'handle_error $LINENO' ERR

# Check command success
if ! pg_dump mydb > backup.sql; then
echo "Backup failed!"
exit 1
fi

# Using || for error handling
pg_dump mydb > backup.sql || {
echo "Backup failed"
exit 1
}

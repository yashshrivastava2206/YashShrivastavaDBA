1. Getting Started with PostgreSQL

  • PostgreSQL architecture and features

2. Database Design Fundamentals

  • Creating databases and schemas
  • Understanding database objects
  • Designing tables and relationships
  • Normalization principles
  • Primary and foreign keys
  • Constraints (CHECK, UNIQUE, NOT NULL)

-- Creating a database
CREATE DATABASE myapp_db;

-- Creating a table with constraints
CREATE TABLE employees (
employee_id SERIAL PRIMARY KEY,
first_name VARCHAR(50) NOT NULL,
last_name VARCHAR(50) NOT NULL,
email VARCHAR(100) UNIQUE,
hire_date DATE DEFAULT CURRENT_DATE,
salary NUMERIC(10,2) CHECK (salary > 0),
department_id INTEGER REFERENCES departments(dept_id)
);

4. SQL Querying

  • SELECT statements and filtering (WHERE)
  • Sorting and limiting results (ORDER BY, LIMIT)
  • Aggregate functions (COUNT, SUM, AVG, MAX, MIN)
  • GROUP BY and HAVING clauses
  • JOIN operations (INNER, LEFT, RIGHT, FULL OUTER)
  • Subqueries and correlated subqueries
  • Common Table Expressions (CTEs)
  • Window functions

-- Complex query with CTE and window function
WITH dept_salaries AS (
SELECT
department_id,
AVG(salary) as avg_salary,
COUNT(*) as emp_count
FROM employees
GROUP BY department_id
)
SELECT
e.employee_id,
e.first_name,
e.last_name,
e.salary,
d.avg_salary,
RANK() OVER (PARTITION BY e.department_id ORDER BY e.salary DESC) as salary_rank
FROM employees e
JOIN dept_salaries d ON e.department_id = d.department_id
WHERE e.salary > d.avg_salary;

-- Creating indexes
CREATE INDEX idx_employees_email ON employees(email);
CREATE INDEX idx_employees_dept ON employees(department_id);

-- Partial index for active employees
CREATE INDEX idx_active_employees
ON employees(last_name)
WHERE status = 'active';

-- Analyzing query performance
EXPLAIN ANALYZE
SELECT * FROM employees
WHERE department_id = 5
ORDER BY hire_date DESC;

using MySql.Data.MySqlClient;

public class DatabaseManager : MonoBehaviour
{
private string connectionString = "Server=localhost;Database=gamedb;Uid=root;Pwd=password;";

public MySqlConnection GetConnection()
{
MySqlConnection connection = new MySqlConnection(connectionString);
try
{
connection.Open();
Debug.Log("Database connected successfully!");
return connection;
}
catch (MySqlException ex)
{
Debug.LogError("Connection failed: " + ex.Message);
return null;
}
}
}

// Insert player data
public void SavePlayer(string username, int score)
{
MySqlConnection conn = GetConnection();
string query = "INSERT INTO players (username, score, created_at) VALUES (@username, @score, NOW())";

MySqlCommand cmd = new MySqlCommand(query, conn);
cmd.Parameters.AddWithValue("@username", username);
cmd.Parameters.AddWithValue("@score", score);

try
{
cmd.ExecuteNonQuery();
Debug.Log("Player saved successfully!");
}
catch (MySqlException ex)
{
Debug.LogError("Save failed: " + ex.Message);
}
finally
{
conn.Close();
}
}
